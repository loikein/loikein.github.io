#### array comparison ####

# my_house greater than or equal to 18
my_house >= 19
# array([ False, True, False, False])

# my_house less than your_house
my_house < your_house
# array([ True,  True, False, False])

# Error!
# my_house >= 19 and my_house < your_house
np.logical_and(my_house >= 19, my_house < your_house)
# array([ False,  True, False, False])

my_house[np.logical_and(my_house >= 19,
                        my_house < your_house)]
# array([ 20.0])

"""Compute the output of a Cobb-Douglas earnings function.
"""
# Define the parameters of the technology.
a = 7.5
gamma_cogn = 2
gamma_noncogn = 3

# Define the values of the inputs.
iq_demeaned = -5
locus_of_control = 3
grit = 7

# Compute the earnings.
earnings = (
    a
    * iq_demeaned ** gamma_cogn
    * ((locus_of_control + grit) / 2) ** gamma_noncogn
)

# Display the results.
print("Annual earnings in Euros:", earnings)

def cobb_douglas(x1, x2, scale, gamma_1, gamma_2):
    """Return the value of a Cobb-Douglas function with
    inputs *x1* and *x2* and technology parameters *scale*,
    *gamma_1*, and *gamma_2*.
    """
    if gamma_2 is None and gamma_1 <= 1:
        gamma_2 = 1 - gamma_1

    out = scale * x1 ** gamma_1 * x2 ** gamma_2
    return out
