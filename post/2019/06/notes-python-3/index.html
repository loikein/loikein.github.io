Writing Functions

#### document every function!!! ####

# default argument: num = 0

def square(num=0):
    """Returns the square of a number."""
    squared_num = num ** 2

    return squared_num

print(square(2))                            # 4
#### return multiple values ####

def power_both(num1, num2):
    """Returns num1 to the power of num2, and num2 to num1."""
    new_num1 = num1 ** num2
    new_num2 = num2 ** num1
    num_tuple = (new_num1, new_num2)

    return num_tuple

power_both(2,3)                             # (8, 9)
#### flexible number of arguments ####

def add_all(*args):
    """Returns the sum of all values."""
    sum_all = 0
    for num in args:
        sum_all += num

    return sum_all

sum_all(2, 3)                               # 5
sum_all(2, 3, 4, 5, 6)                      # 20


def print_all(**kwargs):
    """Print key-value pairs."""
    for k, v in kwargs.items():
        print(k+" is "+v)

print_all(today="sunday", tomorrow="monday")
# results:
# today is sunday
# tomorrow is monday
#### interact with global variables ####

# Create a string: team
team = "teen titans"

# Define change_team()
def change_team():
    """Change the value of the global variable team."""

    # Use team in global scope
    global team

    # Change the value of team in global: team
    team = "justice league"

change_team()

# Print team
print(team)                                 # "justice league"
#### nested functions & returning functions ####

# use nonlocal x to create/change one-scope-above variable

def power_n(n):
    """Return a function of power to the n times."""

    def inner(x):
        powered = x ** n
        return powered

    return inner

square = power_n(2)
cube = power_n(3)

square(2)                                   # 4
cube(2)                                     # 8

Comparison

import numpy as np

#### array comparison ####

my_house = np.array([18.0, 20.0, 10.75, 9.50])
your_house = np.array([14.0, 24.0, 14.25, 9.0])

# my_house greater than or equal to 18
my_house >= 19
# array([ False, True, False, False])

# my_house less than your_house
my_house < your_house
# array([ True,  True, False, False])

# Error!
# my_house >= 19 and my_house < your_house
np.logical_and(my_house >= 19, my_house < your_house)
# array([ False,  True, False, False])

my_house[np.logical_and(my_house >= 19,
                        my_house < your_house)]
# array([ 20.0])

def cobb_douglas(x1, x2, scale, gamma_1, gamma_2):
    """Return the value of a Cobb-Douglas function with
    inputs *x1* and *x2* and technology parameters *scale*,
    *gamma_1*, and *gamma_2*.
    """
    if gamma_2 is None and gamma_1 <= 1:
        gamma_2 = 1 - gamma_1

    out = scale * x1 ** gamma_1 * x2 ** gamma_2
    return out

"""Compute the output of a Cobb-Douglas earnings function.
"""

from cobb_douglas import cobb_douglas

# Define the parameters of the technology.
a = 0.5
gamma_cogn = 1.3
gamma_noncogn = 3

# Set up the data.
data = [
    {"cogn": {"iq_raw": 95}, "noncogn": {"locus of control": 3, "grit": 7, "social": 3}},
    {"cogn": {"iq_raw": 105}, "noncogn": {"locus of control": 2, "grit": 5}},
    {"cogn": {"iq_raw": 100}, "noncogn": {"locus of control": 7, "grit": 1}},
]

def calculate_skill(measures):
    """Given a dictionary *measures*, calculate the value of a skill as the average of its values.its
    """
    my_sum = 0.0
    for m in measures.values():
        my_sum += m

    return my_sum / len(measures)

for obs in data:

    cogn = calculate_skill(obs["cogn"])
    noncogn = calculate_skill(obs["noncogn"])

        cobb_douglas(
            x1=cogn, x2=noncogn, scale=a, gamma_1=gamma_cogn, gamma_2=gamma_noncogn,
        )

# Display the results.
