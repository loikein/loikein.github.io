8.5 BMIの計算・判定

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

int main(void)
{
	int height1;
	double height, weight, BMI, std = 0;
	printf("身長(cm)を入力してください\n");
	scanf("%d", &height1);

	height = height1 / 100.0;

	printf("体重(kg)を入力してください\n");
	scanf("%lf", &weight);

	BMI = weight / height / height;
	std = height * height * 22;

	if (BMI < 18.5) {
		printf("体重は標準より低いです\n");
	} else if (BMI < 25) {
		printf("体重は標準範囲です\n");
	} else {
		printf("体重は標準より高いです\n");
	}
	printf("BMI = %3.1f（標準体重は%4.2fです）\n", BMI, std);

	return 0;
}

21.4 𝜋計算機と.csv出力

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

//generate random numbers
double rnd(void) {
	double ransu;
	ransu = (double)rand()/(RAND_MAX + 0.1);
	return ransu;
}

int main(void) {
	srand((unsigned int)time(NULL));

	double data[2];
	double pi = 4.0;
	int inside = 0;
	int outside = 0;
	int i;

////create file reader, exception, close file
	FILE *fp;
	fp = fopen("pi.csv", "w");

	if (fp == NULL) {
		printf("pi.csvのオープンに失敗しました\n");
		return -1;
	}
	
	printf("pi.csvのオープンに成功しました\n");
	printf("pi.csvにシミュレーション結果を書き込みます\n");
	
//repeat 500 times
	for (i = 0; i < 500; i++) {
		data[0] = rnd();
		data[1] = rnd();

//whether is in the fan
		if(data[0] * data[0] + data[1] * data[1] <= 1.0) {
			inside++;
		} else {
			outside++;
		}
		printf("%d - x = %f, y = %f, inside = %d, outside = %d\n", i + 1, data[0], data[1], inside, outside);

//re-calc pi
		pi = inside * 4.0 / (inside + outside);

//write # of times + pi into pi.csv file
		fprintf(fp, "%d, %f,\n", i + 1, pi);
	}

	fclose(fp);
	return 0;
}
